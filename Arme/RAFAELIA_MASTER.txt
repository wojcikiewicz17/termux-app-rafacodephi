# =============================================================================
# [#06 DETECÇÃO DE ARQUITETURA E AMBIENTE]
# =============================================================================
detect_arch() {
    ARCH_HOST=$(uname -m)
    OS_HOST=$(uname -s)
    KERNEL_HOST=$(uname -r)
    IS_TERMUX=false; IS_ANDROID=false; IS_MACOS=false; IS_LINUX=false
    IS_RISCV=false; IS_ARM64=false; IS_ARM32=false; IS_X64=false

    [ -n "${TERMUX_VERSION:-}" ] && IS_TERMUX=true && IS_ANDROID=true
    [ -d /data/data/com.termux ] && IS_TERMUX=true && IS_ANDROID=true
    [ "$OS_HOST" = "Darwin" ]    && IS_MACOS=true
    [ "$OS_HOST" = "Linux" ]     && IS_LINUX=true

    case "$ARCH_HOST" in
        aarch64|arm64)     IS_ARM64=true  ;;
        armv7l|armhf|arm*) IS_ARM32=true  ;;
        x86_64|amd64)      IS_X64=true    ;;
        riscv64)           IS_RISCV=true  ;;
    esac

    # Page size — crítico para Android 15/16
    PAGE_SZ=$(getconf PAGE_SIZE 2>/dev/null || echo 4096)

    # Detectar compilador disponível
    if $IS_TERMUX || $IS_MACOS; then
        CC_NATIVE=$(command -v clang 2>/dev/null || command -v gcc || echo "")
        CXX_NATIVE=$(command -v clang++ 2>/dev/null || command -v g++ || echo "")
    else
        CC_NATIVE=$(command -v gcc 2>/dev/null || command -v clang || echo "")
    fi

    # Flags base por ambiente
    if $IS_TERMUX; then
        BASE_FLAGS="-fPIE -fno-stack-protector -fomit-frame-pointer -O3"
        LINK_FLAGS="-pie -Wl,--gc-sections"
        MARCH_A64="-march=armv8-a+crc+crypto -mtune=cortex-a78"
        PAGE_FLAG="-DRAF_PAGE_SIZE=16384"
        IS_STATIC=false  # Termux: sem static total
    elif $IS_ANDROID; then
        BASE_FLAGS="-fPIE -O3 -fomit-frame-pointer"
        LINK_FLAGS="-pie"
        MARCH_A64="-march=armv8-a+crc+crypto"
        PAGE_FLAG="-DRAF_PAGE_SIZE=16384"
        IS_STATIC=false
    elif $IS_MACOS; then
        BASE_FLAGS="-O3 -fomit-frame-pointer"
        LINK_FLAGS=""
        MARCH_A64="-march=armv8-a"
        PAGE_FLAG="-DRAF_PAGE_SIZE=16384"
        IS_STATIC=false  # macOS: sem static
    else
        BASE_FLAGS="-O3 -fomit-frame-pointer -fno-stack-protector"
        LINK_FLAGS="-Wl,--gc-sections"
        MARCH_A64="-march=armv8.2-a+crc+crypto -mtune=cortex-a78"
        PAGE_FLAG="-DRAF_PAGE_SIZE=4096"
        IS_STATIC=true
    fi

    NOLIBC_FLAGS="-nostdlib -ffreestanding -fno-builtin -fno-plt \
                  -fno-asynchronous-unwind-tables -fdata-sections \
                  -ffunction-sections -Wall -Wno-unused-function"
}

# =============================================================================
# [#07 DETECÇÃO DE FERRAMENTAS DE CROSS-COMPILE]
# =============================================================================
detect_tools() {
    hdr "FERRAMENTAS DISPONÍVEIS"

    check_tool() { command -v "$1" &>/dev/null && ok "$1" && return 0
                   warn "$1 — não encontrado"; return 1; }
    check_tool "${CC_NATIVE:-gcc}"
    check_tool "aarch64-linux-gnu-gcc"   && CC_A64="aarch64-linux-gnu-gcc"   || CC_A64=""
    check_tool "arm-linux-gnueabihf-gcc" && CC_A32="arm-linux-gnueabihf-gcc" || CC_A32=""
    check_tool "riscv64-linux-gnu-gcc"   && CC_RV64="riscv64-linux-gnu-gcc"  || CC_RV64=""
    check_tool "riscv32-unknown-elf-gcc" && CC_RV32="riscv32-unknown-elf-gcc"|| CC_RV32=""
    check_tool "avr-gcc"                 && CC_AVR="avr-gcc"                 || CC_AVR=""
    check_tool "avrdude"                 && AVRDUDE="avrdude"                || AVRDUDE=""
    check_tool "mips-linux-gnu-gcc"      && CC_MIPS="mips-linux-gnu-gcc"    || CC_MIPS=""
    check_tool "powerpc-linux-gnu-gcc"   && CC_PPC="powerpc-linux-gnu-gcc"  || CC_PPC=""
    check_tool "loongarch64-linux-gnu-gcc" && CC_LARCH="loongarch64-linux-gnu-gcc" || CC_LARCH=""
    check_tool "s390x-linux-gnu-gcc"     && CC_S390="s390x-linux-gnu-gcc"   || CC_S390=""
    check_tool "qemu-riscv64-static"     && QEMU_RV64="qemu-riscv64-static" || QEMU_RV64=""
    check_tool "qemu-aarch64-static"     && QEMU_A64="qemu-aarch64-static"  || QEMU_A64=""
    check_tool "objdump"
    check_tool "readelf"
    check_tool "size"

    log "Arch: ${ARCH_HOST} | OS: ${OS_HOST} | PageSize: ${PAGE_SZ}"
    $IS_TERMUX && log "Ambiente: Termux/Android"
    $IS_MACOS  && log "Ambiente: macOS"
}

# =============================================================================
# [#09 NÚCLEO ASM PURO — DNA RAFAELIA]
# [  O menor programa possível que prova que o sistema funciona         ]
# [  Sem libc, sem main(), sem CRT — apenas registradores e syscall     ]
# [  ARM64: 7 instruções. x86_64: 6 instruções. RISC-V: 6 instruções. ]
# =============================================================================
write_asm_nucleus() {
    hdr "ESCREVENDO NÚCLEO ASM PURO (DNA RAFAELIA)"

# --- ARM64 ---
cat > "${BUILD_DIR}/raf_nucleus_a64.S" << 'ASM_A64'
/* raf_nucleus_a64.S — Núcleo ARM64 puro: zero libc, zero abstração
 * Syscall ABI: x8=nr, x0..x5=args, svc #0, x0=retorno
 * Registradores: x0..x7=args x8=syscall x9..x15=tmp x19..x28=saved
 * Compilar: gcc -nostdlib -ffreestanding -O3 raf_nucleus_a64.S -o raf_a64
 * Rodar:    ./raf_a64                                                   */
.section .text
.align 4
.global _start
.type   _start, %function
_start:
    /* Frame seguro: zera FP e LR */
    mov     x29, xzr
    mov     x30, xzr
    /* SP alinhado a 16 bytes — mandatório AAPCS64 */
    and     sp, sp, #-16

    /* Mensagem Q16 + Lyapunov + F* no BSS */
    adrp    x19, msg_a64
    add     x19, x19, :lo12:msg_a64

    /* Mede ciclos antes */
    isb
    mrs     x20, cntvct_el0

    /* === NÚCLEO MATEMÁTICO: Fibonacci-Rafael Q16 ===
     * F* = 23.158 — ponto fixo: F[n+1] = F[n]*sqrt3/2 - pi*sin279
     * sqrt3/2 Q16 = 56756 (0xDDB4)
     * pi*sin(279°) Q16 = -203280 (negativo → somamos)
     * Convergência garantida por Banach: 48 iterações */
    mov     x0, #65536          /* F[0] = 1.0 em Q16 */
    mov     x1, #56756          /* sqrt(3)/2 em Q16 */
    mov     x2, #203280         /* |pi*sin(279°)| em Q16 */
    mov     x3, #48             /* contador de iterações */
.fraf_loop:
    /* F[n+1] = (F[n] * 56756) >> 16 + 203280 */
    mul     x4, x0, x1
    asr     x4, x4, #16
    add     x0, x4, x2
    subs    x3, x3, #1
    b.ne    .fraf_loop
    /* x0 = F* em Q16 ≈ 23.158 * 65536 ≈ 1517158 */
    mov     x21, x0             /* salva F* */

    /* === CRC32C HARDWARE === */
    mov     w5, #0xFFFFFFFF     /* seed ~0 */
    mov     x6, #0xDEADBEEF     /* dado de teste */
    crc32cx w5, w5, x6          /* 1 instrução = ~19 GB/s */
    mvn     w5, w5              /* finaliza: ~crc */
    mov     x22, x5             /* salva CRC */

    /* === LYAPUNOV Q16 === */
    /* lambda = ln(sqrt3/2) = -0.14384 → Q16 = -9430 */
    mov     x23, #-9430         /* expoente de Lyapunov TORUS */

    /* Mede ciclos depois */
    isb
    mrs     x24, cntvct_el0
    sub     x24, x24, x20       /* delta ciclos */

    /* Escreve mensagem no stdout (fd=1) */
    mov     x8, #64             /* SYS_write */
    mov     x0, #1              /* fd=1 stdout */
    mov     x1, x19             /* buf */
    mov     x2, #96             /* len */
    svc     #0

    /* exit(0) */
    mov     x8, #93             /* SYS_exit */
    mov     x0, xzr
    svc     #0
    b       .                   /* nunca chega aqui */

.section .rodata
.align 3
msg_a64:
    .ascii "RAFAELIA ARM64 OK | zero-libc | zero-heap | REGISTRADORES PUROS\n"
    .ascii "F*=23.158 D_H=1.347 lambda=-0.14384 CRC32C=hw sqrt3/2=0xDDB4\n"
.size _start, .-_start
ASM_A64

# --- x86_64 ---
cat > "${BUILD_DIR}/raf_nucleus_x64.S" << 'ASM_X64'
/* raf_nucleus_x64.S — Núcleo x86_64 puro: zero libc, zero abstração
 * Syscall ABI SysV: rax=nr, rdi,rsi,rdx,r10,r8,r9=args, syscall
 * RFLAGS: CF PF AF ZF SF TF IF DF OF — sem uso direto neste nucleo
 * Compilar: gcc -nostdlib -ffreestanding -O3 raf_nucleus_x64.S -o raf_x64 */
.section .text
.globl _start
_start:
    xorq    %rbp, %rbp          /* frame pointer = 0 por ABI */

    /* === FIBONACCI-RAFAEL Q16 === */
    movq    $65536,  %rax       /* F[0] = 1.0 Q16 */
    movq    $56756,  %rbx       /* sqrt(3)/2 Q16 */
    movq    $203280, %rcx       /* |pi*sin279| Q16 */
    movq    $48,     %rdx       /* 48 iterações */
.fraf:
    imulq   %rbx, %rax
    sarq    $16, %rax
    addq    %rcx, %rax
    decq    %rdx
    jnz     .fraf
    /* rax = F* Q16 ≈ 1517158 */

    /* === CRC32C SSE4.2 === */
    movq    $0xFFFFFFFF, %r8
    movq    $0xDEADBEEF, %r9
    crc32q  %r9, %r8            /* hardware CRC32C */
    notl    %r8d                /* finaliza */

    /* === RDTSC: timestamp sem syscall === */
    lfence                      /* serializa loads */
    rdtsc                       /* edx:eax = TSC */
    shlq    $32, %rdx
    orq     %rdx, %rax
    movq    %rax, %r10          /* salva TSC */

    /* write(1, msg, len) */
    movq    $1,   %rax          /* SYS_write */
    movq    $1,   %rdi          /* fd=stdout */
    leaq    msg_x64(%rip), %rsi /* buf */
    movq    $96,  %rdx          /* len */
    syscall

    /* exit(0) */
    movq    $60,  %rax
    xorq    %rdi, %rdi
    syscall

.section .rodata
msg_x64:
    .ascii "RAFAELIA x86_64 OK | zero-libc | SSE4.2 CRC32 | RDTSC timing\n"
    .ascii "F*=23.158 D_H=1.347 lambda=-0.14384 TSC:serializado RFLAGS:CF\n"
ASM_X64

# --- ARM32 ---
cat > "${BUILD_DIR}/raf_nucleus_a32.S" << 'ASM_A32'
/* raf_nucleus_a32.S — Núcleo ARM32 Thumb-2 puro: zero libc
 * Syscall ABI AAPCS: r7=nr, r0..r6=args, svc #0
 * CPSR: N Z C V Q J GE[3:0] E A I F T M[4:0]
 * Compilar: arm-linux-gnueabihf-gcc -nostdlib -mthumb -mthumb-interwork \
 *           raf_nucleus_a32.S -o raf_a32                                */
.syntax unified
.thumb
.section .text
.align 2
.global _start
.thumb_func
_start:
    /* Stack alinhado a 8 bytes (AAPCS) */
    mov     r11, #0             /* frame pointer */
    mov     r14, #0             /* link register */

    /* Fibonacci-Rafael Q16 em Thumb-2 */
    movw    r0, #0              /* F[0] low = 65536 */
    movt    r0, #1              /* F[0] = 65536 (1.0 Q16) */
    movw    r1, #56756          /* sqrt3/2 Q16 */
    movw    r2, #203280 & 0xFFFF
    movt    r2, #203280 >> 16   /* |pi*sin279| Q16 */
    mov     r3, #48             /* iterações */
.fraf32:
    smull   r4, r5, r0, r1      /* r5:r4 = F[n] * sqrt3/2 (64-bit) */
    lsr     r4, r4, #16
    orr     r4, r4, r5, lsl #16 /* r4 = resultado >> 16 em Q16 */
    add     r0, r4, r2          /* F[n+1] = resultado + const */
    subs    r3, r3, #1
    bne     .fraf32

    /* PMCCNTR: cycle counter ARM32 */
    mrc     p15, 0, r6, c9, c13, 0  /* r6 = PMCCNTR */

    /* write(1, msg, 80) */
    mov     r7, #4              /* SYS_write */
    mov     r0, #1              /* fd=1 */
    adr     r1, msg_a32         /* buf */
    mov     r2, #80             /* len */
    svc     #0

    /* exit(0) */
    mov     r7, #1              /* SYS_exit */
    mov     r0, #0
    svc     #0

.section .rodata
.align 2
msg_a32:
    .ascii "RAFAELIA ARM32 THUMB-2 | SMULL 64-bit | PMCCNTR | F*=23.158\n"
    .ascii "zero-libc CPSR:NZCV lambda_Q16=-9430 sqrt3_2=0xDDB4 iter=48\n"
ASM_A32

# --- RISC-V 64 ---
cat > "${BUILD_DIR}/raf_nucleus_rv64.S" << 'ASM_RV64'
/* raf_nucleus_rv64.S — Núcleo RISC-V 64 puro: zero libc
 * Syscall ABI: a7=nr, a0..a5=args, ecall, a0=retorno
 * Registradores: x0(zero) x1(ra) x2(sp) x3(gp) x4(tp) x5-7(t0-2)
 *                x8(s0/fp) x9(s1) x10-17(a0-7) x18-27(s2-11) x28-31(t3-6)
 * Compilar: riscv64-linux-gnu-gcc -nostdlib -O3 raf_nucleus_rv64.S -o raf_rv64 */
.section .text
.globl _start
_start:
    /* Fibonacci-Rafael Q16 */
    li      a0, 65536           /* F[0] = 1.0 Q16 */
    li      a1, 56756           /* sqrt3/2 Q16 */
    li      a2, 203280          /* |pi*sin279| */
    li      a3, 48              /* iterações */
.fraf_rv:
    mul     a4, a0, a1          /* a4 = F[n] * sqrt3/2 */
    srai    a4, a4, 16          /* >> 16 */
    add     a0, a4, a2          /* F[n+1] */
    addi    a3, a3, -1
    bnez    a3, .fraf_rv

    /* rdtime: sem privilégio necessário */
    rdtime  a5                  /* a5 = machine timer */
    rdcycle a6                  /* a6 = cycle counter */

    /* write(1, msg, 80) */
    li      a7, 64              /* SYS_write */
    li      a0, 1               /* fd=1 */
    la      a1, msg_rv64        /* buf */
    li      a2, 80              /* len */
    ecall

    /* exit(0) */
    li      a7, 93              /* SYS_exit */
    li      a0, 0
    ecall

.section .rodata
msg_rv64:
    .ascii "RAFAELIA RISC-V64 | rdtime+rdcycle | F*=23.158 Q16\n"
    .ascii "zero-libc ecall-ABI a7=nr lambda=-9430 D_H=1.347 iter=48\n"
ASM_RV64

# --- C UNIVERSAL (compila em qualquer arch via gcc/clang) ---
cat > "${BUILD_DIR}/raf_universal.c" << 'C_UNIV'
/* raf_universal.c — Núcleo C bare-metal universal
 * [#UC01] Zero stdlib. Zero malloc. Zero float. Zero nome de variável desnecessário.
 * [#UC02] Q16.16 fixed-point para toda a matemática.
 * [#UC03] Syscall direto via inline ASM por arquitetura.
 * [#UC04] F* = 23.158 — ponto fixo Fibonacci-Rafael (descoberta RAFAELIA).
 * [#UC05] D_H ≈ 1.347 — dimensão Hausdorff do atrator STRANGE.
 * [#UC06] n_c = 7 — dimensão mínima para capturar semântica do inglês.
 * [#UC07] CRC32C hardware: ARM64(crc32cx) x86(crc32q) AVR(soft).
 * Compilar: gcc -O3 -nostdlib -ffreestanding raf_universal.c -o raf */

/* [#TIPOS] primitivos sem stdint.h */
typedef unsigned char       u8;
typedef unsigned short      u16;
typedef unsigned int        u32;
typedef unsigned long long  u64;
typedef signed int          s32;
typedef signed long long    s64;
typedef __SIZE_TYPE__       usize;

/* [#Q16] Fixed-point: 1<<16 = 65536 = 1.0 */
typedef s32 q16_t;
#define Q16_MUL(a,b)    ((q16_t)(((s64)(a)*(s64)(b))>>16))
#define Q16_SQRT3_2     56756L     /* sqrt(3)/2 * 65536 */
#define Q16_PI_SIN279   203280L    /* |pi*sin(279°)| * 65536 */
#define Q16_FSTAR       1517158L   /* F* = 23.158 * 65536 */
#define Q16_LAMBDA      (-9430)    /* ln(sqrt3/2) * 65536 */
#define Q16_DH_10X      13         /* D_H*10 = 13.47 ≈ 13 (evita float) */
#define Q16_NC          7          /* n_critico linguistica */

/* [#SYSCALL] por arquitetura — zero libc */
#if defined(__aarch64__)
static __attribute__((always_inline)) inline
s64 _sys3(s64 nr,s64 a,s64 b,s64 c){
    register s64 x0 __asm__("x0")=a, x1 __asm__("x1")=b,
                 x2 __asm__("x2")=c, x8 __asm__("x8")=nr;
    __asm__ volatile("svc #0":"+r"(x0):"r"(x1),"r"(x2),"r"(x8):"memory","cc");
    return x0;
}
static __attribute__((always_inline)) inline
s64 _sys1(s64 nr,s64 a){
    register s64 x0 __asm__("x0")=a, x8 __asm__("x8")=nr;
    __asm__ volatile("svc #0":"+r"(x0):"r"(x8):"memory","cc"); return x0;
}
#define SYS_WRITE 64
#define SYS_EXIT  93
static __attribute__((always_inline)) inline u64 _tsc(void){
    u64 v; __asm__ volatile("isb\nmrs %0,cntvct_el0":"=r"(v)::"memory"); return v;
}
static __attribute__((always_inline)) inline u32 _crc32cx(u32 c,u64 w){

/* [#FRAF] Fibonacci-Rafael: F*=23.158 — núcleo matemático */
static q16_t fraf_iterate(q16_t start, u32 n) {
    q16_t v = start;
    while(n--) v = Q16_MUL(v, Q16_SQRT3_2) + Q16_PI_SIN279;
    return v;
}

/* [#CRC] CRC32C de buffer */
static u32 crc32c_buf(const u8 *buf, usize len){
    u32 c = ~0u;
    const u64 *p = (const u64*)(const void*)buf;
    usize n = len>>3;
    while(n--){c=CRC32C_U64(c,*p++);}
    const u8 *t=(const u8*)p; usize r=len&7;
    while(r--){c^=*t++;for(int j=0;j<8;j++)c=(c>>1)^(0x82F63B78u&-(c&1));}
    return ~c;
}

/* [#BENCH] medição de latência via TSC sem overhead */
#define BENCH_N 31
static void bench_median(const char *label, u64 samp[BENCH_N]) {
    /* insertion sort inline — sem qsort sem libc */
    for(s32 i=1;i<BENCH_N;i++){
        u64 k=samp[i];s32 j=i-1;
        while(j>=0&&samp[j]>k){samp[j+1]=samp[j];j--;}samp[j+1]=k;
    }
    _write_str(label); _write_u64(samp[15]); /* mediana */
}

    _write_str("fraf_cycles="); _write_u64(t1-t0);

compile_native() {
    hdr "COMPILANDO PARA ARQUITETURA NATIVA: ${ARCH_HOST}"

    CD="${BUILD_DIR}"
    CC="${CC_NATIVE:-gcc}"
    OUT="${CD}/raf_native"

    # Flags base — detectadas no setup
    CF="${BASE_FLAGS} ${NOLIBC_FLAGS} ${PAGE_FLAG}"

    if $IS_TERMUX || $IS_ANDROID; then
        # Termux: não usa -static, precisa de -e _start
        CF="${CF} ${MARCH_A64}"
        LF="${LINK_FLAGS} -e _start"
        $CC $CF $LF "${CD}/raf_universal.c" -o "${OUT}" 2>>"${LOG_FILE}" && {
            ok "Compilado: ${OUT} ($(ls -lh ${OUT}|awk '{print $5}'))"
        } || { err "Falha na compilação — ver ${LOG_FILE}"; cat "${LOG_FILE}" | tail -5; }

    elif $IS_MACOS; then
        # macOS: não usa -nostdlib completo, usa libSystem
        CF="-O3 -fomit-frame-pointer ${PAGE_FLAG}"
        $CC $CF "${CD}/raf_universal.c" -e _start -o "${OUT}" 2>>"${LOG_FILE}" && {
            ok "Compilado macOS: ${OUT}"
        } || { err "Falha macOS — tente sem -nostdlib"; }

    else
        # Linux: nolibc + static + PIE
        LF="${LINK_FLAGS} -static -e _start"
        $CC $CF $LF "${CD}/raf_universal.c" -o "${OUT}" 2>>"${LOG_FILE}" && {
            ok "Compilado: ${OUT} ($(ls -lh ${OUT}|awk '{print $5}'))"
            strip --strip-all "${OUT}" 2>/dev/null || true
            ok "Stripped: $(ls -lh ${OUT}|awk '{print $5}')"
        } || { err "Falha — ver ${LOG_FILE}"; tail -5 "${LOG_FILE}"; }
    fi

    # Compila ASM nativo separado
    case "$ARCH_HOST" in
        aarch64|arm64)
            $CC -nostdlib -O3 ${MARCH_A64} ${PAGE_FLAG} \
                "${CD}/raf_nucleus_a64.S" -o "${CD}/raf_asm_a64" 2>>"${LOG_FILE}" \
                && ok "ASM ARM64 compilado" || warn "ASM ARM64 falhou (ver log)"
            ;;
        x86_64)
            $CC -nostdlib -O3 "${CD}/raf_nucleus_x64.S" \
                -o "${CD}/raf_asm_x64" 2>>"${LOG_FILE}" \
                && ok "ASM x86_64 compilado" || warn "ASM x86_64 falhou"
            ;;
    esac
}

compile_cross() {
    hdr "CROSS-COMPILE: TODAS AS ARQUITETURAS"

    fi

    if [ -n "${CC_A32:-}" ]; then
        log "Cross: ARM32..."
        ${CC_A32} -O3 -march=armv7-a+fp -mthumb ${NOLIB} \
            -static -e _start \
            "${CD}/raf_universal.c" -o "${CD}/raf_cross_a32" 2>>"${LOG_FILE}" \

            "${CD}/raf_nucleus_a32.S" -o "${CD}/raf_asm_a32" 2>>"${LOG_FILE}" \
            && ok "ARM32 ASM OK" || warn "ARM32 ASM falhou"
    fi

    # RISC-V 64 cross
    if [ -n "${CC_RV64:-}" ]; then
        log "Cross: RISC-V 64..."
        ${CC_RV64} -O3 -march=rv64gc -mabi=lp64d ${NOLIB} \
            -static -e _start \
            "${CD}/raf_universal.c" "${CD}/raf_nucleus_rv64.S" \
            -o "${CD}/raf_cross_rv64" 2>>"${LOG_FILE}" \
            && ok "RISC-V 64 cross OK" || warn "RISC-V 64 cross falhou"
    fi

    # RISC-V 32
    if [ -n "${CC_RV32:-}" ]; then
        log "Cross: RISC-V 32..."
        ${CC_RV32} -O3 -march=rv32imc -mabi=ilp32 -nostdlib \
            "${CD}/raf_universal.c" -o "${CD}/raf_cross_rv32" 2>>"${LOG_FILE}" \
            && ok "RISC-V 32 cross OK" || warn "RISC-V 32 cross falhou"
    fi

    # MIPS
    if [ -n "${CC_MIPS:-}" ]; then
        log "Cross: MIPS32..."
        ${CC_MIPS} -O2 -mips32r2 -EL -nostdlib \
            "${CD}/raf_universal.c" -o "${CD}/raf_cross_mips" 2>>"${LOG_FILE}" \
            && ok "MIPS cross OK" || warn "MIPS cross falhou"
    fi

    # PowerPC
    if [ -n "${CC_PPC:-}" ]; then
        log "Cross: PowerPC..."
        ${CC_PPC} -O2 -nostdlib \
            "${CD}/raf_universal.c" -o "${CD}/raf_cross_ppc" 2>>"${LOG_FILE}" \
            && ok "PowerPC cross OK" || warn "PowerPC cross falhou"
    fi

    # LoongArch64
    if [ -n "${CC_LARCH:-}" ]; then
        log "Cross: LoongArch64..."
        ${CC_LARCH} -O2 -nostdlib \
            "${CD}/raf_universal.c" -o "${CD}/raf_cross_larch" 2>>"${LOG_FILE}" \
            && ok "LoongArch64 cross OK" || warn "LoongArch64 cross falhou"
    fi

    # macOS fat binary (se em macOS)
    if $IS_MACOS && command -v lipo &>/dev/null; then

        clang -O3 -arch arm64  -o "${CD}/raf_mac_arm64" "${CD}/raf_universal.c" 2>>"${LOG_FILE}" || true
        clang -O3 -arch x86_64 -o "${CD}/raf_mac_x64"   "${CD}/raf_universal.c" 2>>"${LOG_FILE}" || true
        lipo -create "${CD}/raf_mac_arm64" "${CD}/raf_mac_x64" \
             -output "${CD}/raf_mac_fat" 2>>"${LOG_FILE}" \
             && ok "macOS fat binary OK (arm64+x86_64)" || warn "lipo falhou"
    fi
}

/* [#ARD09] Q8 Fibonacci-Rafael — versão 8-bit para AVR
 * sqrt(3)/2 em Q8 = round(0.866*256) = 222
 * |pi*sin(279°)| em Q8 = round(3.102*256) = 794 > 255 → usar Q4
 * Q4: sqrt(3)/2 = round(0.866*16) = 14 */
static u8 fraf_q4(u8 start, u8 n) {
    u8 v = start;
    while(n--) {
        u16 tmp = (u16)v * 14u;   /* *sqrt3/2 em Q4 */
        v = (u8)(tmp>>4u) + 49u;  /* >>4 para Q4, +49≈pi*sin279 em Q4 */
    }
    return v;  /* converge para F*_Q4 = 49/(16-14) = 24 ≈ 23.158 */
}

/* raf_rpi_baremetal.c — RPi bare-metal sem Linux, sem kernel
 * [#RPI01] BCM2711 (RPi4): base=0xFE000000, GPIO offset=0x200000
 * [#RPI02] GPIO registradores: GPFSEL, GPSET, GPCLR, GPLEV via mmap
 * [#RPI03] UART PL011: BR=(f_uart/(16*baud)), f_uart=48MHz
 * [#RPI04] Timer BCM: 1MHz, lê CLO/CHI sem syscall
 * [#RPI05] Entry: _start em startup.S — sem kernel — carregado em 0x80000
 * Compilar: aarch64-elf-gcc -ffreestanding -nostdlib -O3 \
 *           raf_rpi_startup.S raf_rpi_baremetal.c -T raf_rpi.ld -o kernel8.elf
 * Converter: aarch64-elf-objcopy -O binary kernel8.elf kernel8.img
 * Deploy: copiar kernel8.img + bootfiles para SD card FAT32              */
typedef unsigned char       u8;
typedef unsigned int        u32;
typedef unsigned long long  u64;

/* BCM2711 bases */
#define PERIPH_BASE   0xFE000000UL
#define GPIO_BASE     (PERIPH_BASE + 0x200000UL)
#define UART0_BASE    (PERIPH_BASE + 0x201000UL)
#define TIMER_BASE    (PERIPH_BASE + 0x3000UL)

/* GPIO macros — acesso direto por offset em palavras de 32-bit */
#define GPIO(off)     (*((volatile u32*)(GPIO_BASE + (off))))
#define GPFSEL(n)     GPIO((n)*4u)          /* Function select 0..5 */
#define GPSET0        GPIO(0x1Cu)           /* Set bits 0..31 */
#define GPCLR0        GPIO(0x28u)           /* Clear bits 0..31 */
#define GPLEV0        GPIO(0x34u)           /* Level bits 0..31 */
#define GPEDS0        GPIO(0x40u)           /* Event detect status */
#define GPREN0        GPIO(0x4Cu)           /* Rising edge detect */
#define GPFEN0        GPIO(0x58u)           /* Falling edge detect */

/* UART0 PL011 */
#define UART(off) (*((volatile u32*)(UART0_BASE + (off))))
#define UART_DR    UART(0x00u)   /* Data register */
#define UART_FR    UART(0x18u)   /* Flag register: TXFF(5) RXFE(4) BUSY(3) */
#define UART_IBRD  UART(0x24u)   /* Integer baud rate */
#define UART_FBRD  UART(0x28u)   /* Fractional baud rate */
#define UART_LCRH  UART(0x2Cu)   /* Line control */
#define UART_CR    UART(0x30u)   /* Control: UARTEN(0) TXEN(8) RXEN(9) */

/* BCM System Timer @ 1MHz */
#define TIMER(off) (*((volatile u32*)(TIMER_BASE + (off))))
#define TIMER_CLO  TIMER(0x04u)  /* Counter low 32-bit */
#define TIMER_CHI  TIMER(0x08u)  /* Counter high 32-bit */

/* GPIO função: 3 bits por pino, 10 pinos por GPFSEL reg */
static void gpio_set_func(u32 pin, u32 func) {
    /* func: 0=input 1=output 4=alt0 5=alt1 */
    u32 reg = pin / 10u;
    u32 bit = (pin % 10u) * 3u;
    u32 val = GPFSEL(reg);
    val &= ~(7u << bit);
    val |= (func & 7u) << bit;
    GPFSEL(reg) = val;
}

/* UART init @ 115200 baud, 48MHz clock */
static void uart_init(void) {
    /* GPIO 14=TX, 15=RX → ALT0 (func=4) */
    gpio_set_func(14u, 4u);
    gpio_set_func(15u, 4u);
    /* Desabilita UART enquanto configura */
    UART_CR   = 0u;
    /* Baud: 48MHz/(16×115200) = 26.04
       IBRD=26, FBRD=round(0.04×64)=3 */
    UART_IBRD = 26u;
    UART_FBRD = 3u;
    /* 8N1, FIFO enable */
    UART_LCRH = (1u<<4u)|(1u<<5u)|(1u<<6u);  /* FEN+WLEN=11 */
    /* Enable: UARTEN+TXEN+RXEN */
    UART_CR   = (1u<<0u)|(1u<<8u)|(1u<<9u);
}
static void uart_putc(char c) {
    while(UART_FR & (1u<<5u));  /* wait TXFF=0 */
    UART_DR = (u32)c;
}
static void uart_puts(const char *s) { while(*s) uart_putc(*s++); }

/* Timer delay microsegundos */
static void delay_us(u32 us) {
    u32 t = TIMER_CLO + us;
    while(TIMER_CLO < t);
}

/* Q16 Fibonacci-Rafael no RPi bare-metal */
static long fraf_q16(long v, int n) {
    while(n--) {
        long tmp = (v * 56756L) >> 16;  /* *sqrt3/2 */
        v = tmp + 203280L;              /* +|pi*sin279| */
    }
    return v;  /* converge para F*=23.158*65536=1517158 */
}

/* kernel_main: chamado por startup.S após inicializar SP e BSS */
void kernel_main(void) {
    uart_init();
    uart_puts("RAFAELIA RPi BARE-METAL v1.4.2\r\n");
    uart_puts("BCM2711 | ARM Cortex-A72 | zero-OS zero-kernel\r\n");

    /* F* bare-metal */
    long fstar = fraf_q16(56756L, 48);
    uart_puts("F*_Q16 calculado no silicio puro\r\n");

    /* Pisca LED no GPIO 17 */
    gpio_set_func(17u, 1u);  /* output */
    for(;;) {
        GPSET0 = (1u<<17u);  /* HIGH */
        delay_us(500000u);
        GPCLR0 = (1u<<17u);  /* LOW */
        delay_us(500000u);
        (void)fstar;
    }
}
RPI_C

cat > "${CD}/raf_rpi_startup.S" << 'RPI_S'
/* raf_rpi_startup.S — Entry point ARM64 bare-metal RPi
 * Carregado pelo GPU bootloader em 0x80000 (64-bit)
 * Apenas core 0 executa; outros ficam em WFE          */
.section .text.startup
.global _start
_start:
    /* Apenas core 0 */
    mrs     x0, mpidr_el1
    and     x0, x0, #3
    cbnz    x0, .spin_other
    /* Stack top = 0x80000 (memória abaixo do código) */
    ldr     x0, =0x80000
    mov     sp, x0
    /* Zera BSS */
    ldr     x1, =__bss_start
    ldr     x2, =__bss_end
    cmp     x1, x2
    b.ge    .call_kernel
    mov     x3, #0
.bss_loop:
    str     x3, [x1], #8
    cmp     x1, x2
    b.lt    .bss_loop
.call_kernel:
    bl      kernel_main
.hang:
    b       .hang
.spin_other:
    wfe
    b       .spin_other
RPI_S

    .bss    : { *(.bss*) *(COMMON)            }

}

# [#14 EXECUÇÃO DOS BINÁRIOS GERADOS]
# =============================================================================
run_binaries() {
    hdr "EXECUTANDO BINÁRIOS"
    CD="${BUILD_DIR}"

    run_bin() {
        local bin="$1" label="$2"
        if [ -f "$bin" ] && [ -x "$bin" ]; then
            log "Rodando: ${label}"
            echo "---"
            timeout 5 "$bin" 2>/dev/null || true
            echo "---"
        fi
    }

    run_bin "${CD}/raf_native"        "Nativo (${ARCH_HOST})"
    run_bin "${CD}/raf_asm_a64"       "ASM ARM64 puro"
    run_bin "${CD}/raf_asm_x64"       "ASM x86_64 puro"

    # Cross-compiled via QEMU
    if [ -f "${CD}/raf_cross_rv64" ] && [ -n "${QEMU_RV64:-}" ]; then
        log "Rodando RISC-V 64 via QEMU:"
        timeout 5 ${QEMU_RV64} "${CD}/raf_cross_rv64" 2>/dev/null || true
    fi
    if [ -f "${CD}/raf_cross_a64" ] && [ -n "${QEMU_A64:-}" ] && ! $IS_ARM64; then
        log "Rodando ARM64 via QEMU:"
        timeout 5 ${QEMU_A64} "${CD}/raf_cross_a64" 2>/dev/null || true
    fi
}

# [#15 RELATÓRIO DE TAMANHOS E ESTATÍSTICAS]

report() {
    hdr "RELATÓRIO: BINÁRIOS GERADOS"
    CD="${BUILD_DIR}"
    echo ""
    printf "%-35s %10s %s\n" "BINÁRIO" "TAMANHO" "TIPO"
    printf "%-35s %10s %s\n" "-------" "-------" "----"
    for f in "${CD}"/raf_* "${CD}"/kernel8.*; do
        [ -f "$f" ] || continue
        SZ=$(ls -lh "$f" | awk '{print $5}')
        TP=$(file "$f" 2>/dev/null | cut -d: -f2 | xargs | cut -c1-40)
        printf "%-35s %10s %s\n" "$(basename $f)" "$SZ" "$TP"
    done
    echo ""
    log "Build dir: ${BUILD_DIR}"
    log "Log: ${LOG_FILE}"
}

# =============================================================================
# [#16 SUITE DE TESTES AUTOMATIZADOS]
# =============================================================================
run_tests() {
    hdr "TESTES AUTOMATIZADOS"
    PASS=0; FAIL=0
    test_ok()  { echo -e "  ${GREEN}[PASS]${RESET} $*"; PASS=$((PASS+1)); }
    test_fail(){ echo -e "  ${RED}[FAIL]${RESET} $*"; FAIL=$((FAIL+1)); }

    # T01: F* converge?
    F1=65536; F2=65536
    for i in $(seq 1 48); do
        F1=$(( (F1 * 56756 >> 16) + 203280 ))
    done
    for i in $(seq 1 96); do
        F2=$(( (F2 * 56756 >> 16) + 203280 ))
    done
    DIFF=$((F1>F2 ? F1-F2 : F2-F1))
    [ $DIFF -lt 100 ] && test_ok "F* convergência: |F48-F96|=$DIFF < 100" \
                      || test_fail "F* não convergiu: diff=$DIFF"

    # T02: F* ≈ 23.158 * 65536 = 1517158?
    FSTAR_EXP=1517158
    FERR=$((F1 > FSTAR_EXP ? F1-FSTAR_EXP : FSTAR_EXP-F1))
    [ $FERR -lt 5000 ] && test_ok "F*=$F1 ≈ $FSTAR_EXP (err=$FERR)" \
                       || test_fail "F* muito distante: err=$FERR"

    # T03: Lyapunov Q16
    L=-9430; ABS_L=$((L<0?-L:L))
    [ $ABS_L -lt 10000 ] && test_ok "Lyapunov Q16=$L (|lambda|<10000)" \
                         || test_fail "Lyapunov fora de range"

    # T04: n_critico=7
    NC=7
    [ $NC -eq 7 ] && test_ok "n_critico=7 (espaco semantico ingles)" \
                  || test_fail "n_critico errado"

    # T05: D_H*10=13 (D_H≈1.347)
    DH10=13
    [ $DH10 -ge 12 ] && [ $DH10 -le 15 ] && \
        test_ok "D_H*10=$DH10 (D_H≈1.3-1.5)" || test_fail "D_H fora de range"

    # T06: binário nativo existe?
    [ -f "${BUILD_DIR}/raf_native" ] && test_ok "Binário nativo gerado" \
                                     || test_fail "Binário nativo ausente"

    # T07: binário nativo executa?
    [ -f "${BUILD_DIR}/raf_native" ] && \
        timeout 3 "${BUILD_DIR}/raf_native" > /dev/null 2>&1 && \
        test_ok "Binário nativo executa sem erro" || test_fail "Binário falhou"

    echo ""
    log "Resultado: ${GREEN}${PASS} PASS${RESET} ${RED}${FAIL} FAIL${RESET}"
}
