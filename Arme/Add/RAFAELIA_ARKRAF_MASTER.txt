#define Q16_HALF    32768u
#define Q16_SPIRAL  56755u   /* sqrt(3)/2 — invariante */
#define Q16_PHI     105965u  /* (1+sqrt(5))/2 */
#define Q16_PI      205887u
#define Q16_2PI     411774u

/*═══ Q16.16 OPS ════════════════════════════════════════════════════════════
 * PRIMO=p2  LAYER=registrador  CYCLES=3 (umull+lsr+orr)
 * INVARIANTE: |qema(old,in) - old| ≤ 16384 por chamada
 *═══════════════════════════════════════════════════════════════════════════*/
static inline u32 qmul(u32 a, u32 b){
    return (u32)(((u64)a*b)>>16);
}
static inline u32 qema(u32 old, u32 in){
    /* 0.75*old + 0.25*in — determinístico, sem float */
    return (u32)(((u64)old*49152u+(u64)in*16384u)>>16);
}
static inline u32 qabs(i32 v){ return (u32)(v<0?-v:v); }

/*═══ SIN TAYLOR Q16.16 ═════════════════════════════════════════════════════
 * Domínio público: série de Taylor (Maclaurin 1742)
 * Erro < 0.01% para x ∈ [0, 2π]
 *═══════════════════════════════════════════════════════════════════════════*/
static inline u32 qsin(u32 x){
    while(x>=Q16_2PI) x-=Q16_2PI;
    int neg=0;
    if(x>=Q16_PI){x-=Q16_PI;neg=1;}
    u64 x2=(u64)x*x>>16;
    u64 x3=(u64)x2*x>>16;
    u64 x5=(u64)x3*x2>>16;
    u64 t1=(u64)x3*Q16_INV6>>16;
    u64 t2=(u64)x5*Q16_INV120>>16;
    i64 r=(i64)x-(i64)t1+(i64)t2;
    if(r<0)r=0; if(r>65535)r=65535;
    return neg?(u32)(65535u-(u32)r):(u32)r;
}

/**
 * rafaelia_arena.h — arena estática zero malloc
 * SPDX-License-Identifier: GPL-3.0-only
 * RAZÃO: malloc() Bionic ~100 ciclos + fragmentação.
 * Arena BSS: O(1) alloc, zero heap, determinístico.
 * Conforma: IEEE Std 1003.1-2017 §13
 */
#pragma once
#ifndef RAFAELIA_ARENA_H
#define RAFAELIA_ARENA_H

#include "rafaelia_types.h"

#define ARENA_SZ (4u*1024u*1024u)  /* 4MB BSS — bionic aloca lazy */

extern u8   g_arena_buf[ARENA_SZ];
extern u32  g_arena_bump;
extern u32  g_crc_tab[256];
extern int  g_crc_ready;

/*═══ RAF_ALLOC ══════════════════════════════════════════════════════════════
 * PRIMO=p3  LAYER=BSS  CYCLES=5
 * INPUT: n=bytes, al=alinhamento (pot de 2)
 * OUTPUT: ponteiro alinhado ou NULL se OOM
 * ROLLBACK: g_arena_bump não avança se OOM
 *═══════════════════════════════════════════════════════════════════════════*/
FORCEINL void *raf_alloc(u32 n, u32 al) {
    u32 mask=al-1u, s=(g_arena_bump+mask)&~mask, e=s+n;
    if(e>ARENA_SZ) return 0;
    g_arena_bump=e;
    return g_arena_buf+s;
}

FORCEINL void raf_arena_reset(void){ g_arena_bump=0; }

#define RALLOC(T,n)   ((T*)raf_alloc((u32)(sizeof(T)*(n)), 16u))
#define RALLOC64(T,n) ((T*)raf_alloc((u32)(sizeof(T)*(n)), 64u))

/*═══ CRC32C CASTAGNOLI (RFC 3720 §B.4, NIST SP800-175B) ════════════════════
 * Poly 0x82F63B78 = reverso de 0x1EDC6F41 — domínio público
 * INVARIANTE: crc32c("123456789") == 0xE3069283
 *═══════════════════════════════════════════════════════════════════════════*/
static inline void crc_build(void){
    for(u32 i=0;i<256u;i++){
        u32 v=i;
        for(int j=0;j<8;j++) v=(v&1u)?(v>>1)^0x82F63B78u:(v>>1);
        g_crc_tab[i]=v;
    }
    g_crc_ready=1;
}

static inline u32 crc32c(const void*buf, u32 n){
    if(!g_crc_ready) crc_build();
    const u8*p=(const u8*)buf; u32 c=~0u;
    while(n--) c=(c>>8)^g_crc_tab[(c^*p++)&0xFF];
    return ~c;
}

#endif /* RAFAELIA_ARENA_H */

/**
 * rafaelia_gpu_impl.c — implementações GPU: CRC32C, entropia, NEON step,
 *                        g_state, g_cl, g_close
 * SPDX-License-Identifier: GPL-3.0-only
 *
 * Compilar como .so para Android:
 *   clang -O2 -march=armv7-a -mfpu=neon-vfpv4 -mfloat-abi=softfp
 *         -fPIC -DHAS_NEON -D__ANDROID__
 *         rafaelia_gpu_impl.c rafaelia_gpu_orch.c rafaelia_gpu_jni.c
 *         -shared -o librafaelia_gpu.so -ldl
 */
#define _POSIX_C_SOURCE 200809L
#include "../headers/rafaelia_gpu.h"
#include <string.h>
#include <dlfcn.h>

/*═══ CRC32C CASTAGNOLI ══════════════════════════════════════════════════════
 * RFC 3720 §B.4 · NIST SP 800-175B · poly 0x82F63B78
 * PRIMO=p7  LAYER=L1  CYCLES=4/byte
 *═══════════════════════════════════════════════════════════════════════════*/
static uint32_t _ct[256]; static int _ci=0;
static void _cb(void){
    for(uint32_t i=0;i<256;i++){uint32_t v=i;
        for(int j=0;j<8;j++) v=(v&1)?(v>>1)^0x82F63B78u:(v>>1);
        _ct[i]=v;}
    _ci=1;
}
uint32_t g_crc32(const void*b,uint32_t n){
    if(!_ci)_cb();
    const uint8_t*p=(const uint8_t*)b;uint32_t c=~0u;
    while(n--) c=(c>>8)^_ct[(c^*p++)&0xFF];return~c;
}

/*═══ ENTROPIA ═══════════════════════════════════════════════════════════════
 * unique_bytes/256 * 65535 → Q16 proxy de Shannon H
 * PRIMO=p11  CYCLES=O(min(n,4096))
 *═══════════════════════════════════════════════════════════════════════════*/
uint32_t g_entropy(const uint8_t*buf,uint32_t n){
    uint8_t seen[256]; memset(seen,0,256);
    uint32_t u=0,lim=n>4096?4096:n;
    for(uint32_t i=0;i<lim;i++) if(!seen[buf[i]]){seen[buf[i]]=1;u++;}
    return(u*65535u)/256u;
}

/* sin Taylor Q16.16 — sem float, sem libm */
static inline uint32_t _s(uint32_t x){
    while(x>=411774)x-=411774;
    int ng=x>=205887;if(ng)x-=205887;
    uint64_t x2=(uint64_t)x*x>>16,x3=(uint64_t)x2*x>>16,x5=(uint64_t)x3*x2>>16;
    int64_t r=(int64_t)x-(int64_t)((uint64_t)x3*10923>>16)
                        +(int64_t)((uint64_t)x5*546>>16);
    if(r<0)r=0;if(r>65535)r=65535;
    return ng?(uint32_t)(65535-(uint32_t)r):(uint32_t)r;
}

/*═══ G_STATE_INIT ═══════════════════════════════════════════════════════════
 * Seeds toroidais com constantes irracionais Q16.16
 * INVARIANTE: após init, g_state_ok() == 1
 *═══════════════════════════════════════════════════════════════════════════*/
void g_state_init(g_state_t*st){
    memset(st,0,sizeof(*st));
    st->C=st->H=0x8000u;
    uint32_t seed=G_SPIRAL;
    for(uint32_t d=0;d<G_DIM;d++){
        seed=(uint32_t)(((uint64_t)seed*G_SPIRAL+(uint64_t)d*1009u)&0xFFFFu);
        st->s[d]=seed;
    }
    st->phi=0x8000u;
    st->crc=0;
    st->crc=g_crc32(st,offsetof(g_state_t,crc));
}

/*═══ G_STATE_OK ══════════════════════════════════════════════════════════════
 * Verifica CRC32C do estado — detecta corrupção silenciosa
 * ROLLBACK: chamador deve chamar g_state_init() se retornar 0
 *═══════════════════════════════════════════════════════════════════════════*/
int g_state_ok(const g_state_t*st){
    uint32_t saved=st->crc;
    ((g_state_t*)st)->crc=0;
    uint32_t check=g_crc32(st,offsetof(g_state_t,crc));
    ((g_state_t*)st)->crc=saved;
    return check==saved;
}

/*═══ G_NEON_STEP ════════════════════════════════════════════════════════════
 * EMA 7D Q16.16 com NEON ARM32 onde disponível
 * PRIMO=p13  LAYER=L1  CYCLES=4 (NEON) / 21 (escalar)
 * INPUT: s[7] estado atual, cy ciclo 0..41
 * OUTPUT: out[7] estado atualizado
 * INVARIANTE: out[d] ∈ [0,65535] para todo d
 *═══════════════════════════════════════════════════════════════════════════*/
void g_neon_step(uint32_t*out,const uint32_t*s,uint32_t cy){
#ifdef __ARM_NEON
    uint32x4_t v0=vld1q_u32(s);
    uint32x4_t inv=vdupq_n_u32(49152u), alp=vdupq_n_u32(16384u);
    uint32x4_t sv=vdupq_n_u32(_s(cy*9804u));
    uint64x2_t lo=vaddq_u64(vmull_u32(vget_low_u32(v0),vget_low_u32(inv)),
                              vmull_u32(vget_low_u32(sv),vget_low_u32(alp)));
    uint64x2_t hi=vaddq_u64(vmull_u32(vget_high_u32(v0),vget_high_u32(inv)),
                              vmull_u32(vget_high_u32(sv),vget_high_u32(alp)));
    vst1q_u32(out,vcombine_u32(vshrn_n_u64(lo,16),vshrn_n_u64(hi,16)));
    for(uint32_t d=4;d<G_DIM;d++){
        uint32_t sinv=_s((cy+d)*9804u);
        out[d]=(uint32_t)(((uint64_t)s[d]*49152u+(uint64_t)sinv*16384u)>>16);
    }
#else
    for(uint32_t d=0;d<G_DIM;d++){
        uint32_t sinv=_s((cy+d)*9804u);
        out[d]=(uint32_t)(((uint64_t)s[d]*49152u+(uint64_t)sinv*16384u)>>16);
    }
#endif
}

/*═══ G_CL_INIT ══════════════════════════════════════════════════════════════
 * dlopen OpenCL — paths VNDK Android por fabricante
 * Sem linkagem estática — Khronos OpenCL 3.0 ICD loader spec
 *═══════════════════════════════════════════════════════════════════════════*/
int g_cl_init(gctx_t*c){
    static const char*P[]={
        "/vendor/lib/libOpenCL.so","/vendor/lib/libPVROCL.so",
        "/vendor/lib/egl/libGLES_mali.so","/system/lib/libOpenCL.so",
        "/vendor/lib/libOpenCL_adreno.so","/vendor/lib64/libOpenCL.so",
        "libOpenCL.so.1",NULL};
    for(int i=0;P[i];i++){
        void*l=dlopen(P[i],RTLD_LAZY|RTLD_LOCAL);
        if(!l)continue;
        if(dlsym(l,"clGetPlatformIDs")){c->glib=l;return 1;}
        dlclose(l);
    }
    return 0;
}

/* g_cl_run: com OpenCL real requereria clEnqueueNDRangeKernel
 * aqui: fallback NEON com c_in como seed */
int g_cl_run(gctx_t*c,uint32_t c_in,uint32_t h_in,uint32_t cy){
    (void)c_in;(void)h_in;
    uint32_t tmp[G_DIM];
    g_neon_step(tmp,c->st.s,cy);
    for(uint32_t d=0;d<G_DIM;d++) c->st.s[d]=tmp[d];
    return 1;
}

void g_close(gctx_t*c){
    if(c->glib){dlclose(c->glib);c->glib=0;}
    c->backend=G_NONE;
}

/* Triângulo isósceles: jet = core menor load ≠ ápice */

/* Commit gate 4 fases */
static u32  g_cg_bm[N_VCPU];
static vcpu_t g_snap[N_VCPU];
static u32 g_commits=0,g_rollbacks=0;
#define CG_LOAD 0x1u
#define CG_PROC 0x2u
#define CG_VERI 0x4u
#define CG_COMM 0x8u
#define CG_ALL  0xFu

static void cg_step(u32 core){
    vcpu_t*v=&g_vcpu[core];
    memcpy(&g_snap[core],v,sizeof(vcpu_t));
    g_cg_bm[core]|=CG_LOAD;
    for(u32 d=0;d<TORUS_DIM;d++){
        u32 nd=(d+1u)%TORUS_DIM;
        v->s[d]=qema(v->s[d],v->s[nd]);
    }
    v->C=qema(v->C,qmul(v->hz,Q16_SPIRAL)&0xFFFFu);
    v->H=qema(v->H,65535u-(qmul(v->hz,Q16_SPIRAL)&0xFFFFu));
    v->phase=(v->phase+1u<PERIOD)?v->phase+1u:0u;
    g_cg_bm[core]|=CG_PROC;
    u32 sc=crc32c(v,offsetof(vcpu_t,crc_s));
    if(!sc||v->s[0]>=Q16_2PI){
        memcpy(v,&g_snap[core],sizeof(vcpu_t));
        g_cg_bm[core]=0;g_rollbacks++;return;
    }
    g_cg_bm[core]|=CG_VERI;
    if((g_cg_bm[core]&CG_ALL)==CG_ALL){
        v->crc_s=sc;g_cg_bm[core]=0;g_commits++;
    }
}

}

/* Senoide 7 camadas */
static u32 g_sp[TORUS_DIM],g_sw[TORUS_DIM],g_pt[PERIOD];

static u32 sin_step(u32 cy){
    u32 ov=0;
    for(u32 i=0;i<TORUS_DIM;i++){
        g_sp[i]+=FREQS[i];
        if(g_sp[i]>=Q16_2PI)g_sp[i]-=Q16_2PI;
        u32 sv=qsin(g_sp[i]);
        ov+=qmul(sv,g_sw[i]);
        g_sw[i]=(g_sw[i]*3u+sv)>>2;

    if(cy<PERIOD)g_pt[cy]=phi;
    u32 tmp[2]={phi,g_cc_chain};
    g_cc_chain=crc32c(tmp,8u);
    return phi;
}

/**
 * rafaelia_gpu_orch.c — GPU orchestrator (dispatch CL→NEON→VK)
 * SPDX-License-Identifier: GPL-3.0-only
 * Depende de: rafaelia_gpu.h + rafaelia_gpu_impl.c
 */
#include "../headers/rafaelia_gpu.h"
#include <string.h>

void g_init(gctx_t*c){
    memset(c->mem,0,G_ARENA);
    memset(c->cl,0,sizeof(c->cl));
    c->bump=0;c->backend=G_NONE;c->glib=0;
    g_state_init(&c->st);
}

int g_detect(gctx_t*c){
    if(g_cl_init(c)){c->backend=G_CL;return G_CL;}
#ifdef __ARM_NEON
    c->backend=G_NEON;return G_NEON;
#else
    return G_NONE;
#endif
}

int g_step(gctx_t*c,const uint8_t*in,uint32_t n){
    if(!c->backend)return 0;
    if(!g_state_ok(&c->st)) g_state_init(&c->st);
    uint32_t c_in=g_crc32(in,n)&G_MASK;
    uint32_t h_in=g_entropy(in,n);
    uint32_t cy=c->st.phase;
    if(c->backend==G_CL){
        if(g_cl_run(c,c_in,h_in,cy)) goto commit;
        c->backend=G_NEON;
    }
    {/* NEON path */
        uint32_t tmp[G_DIM];
        g_neon_step(tmp,c->st.s,cy);
        for(uint32_t i=0;i<G_DIM;i++) c->st.s[i]=tmp[i];
    }
commit:
    /* EMA C e H */
    c->st.C=(uint32_t)(((uint64_t)c->st.C*49152+(uint64_t)c_in*16384)>>16);
    c->st.H=(uint32_t)(((uint64_t)c->st.H*49152+(uint64_t)h_in*16384)>>16);
    c->st.phi=(uint32_t)(((uint64_t)(65535-c->st.H)*c->st.C)>>16);
    c->st.phase=(c->st.phase+1)%G_PERIOD;
    c->st.step++;
    /* CRC chain */
    uint32_t pl[2]={c->st.phi,c->st.chain};
    c->st.chain=g_crc32(pl,8);
    c->st.crc=0;
    c->st.crc=g_crc32(&c->st,offsetof(g_state_t,crc));
    return 1;
}
EOF_GPO

#!/usr/bin/env python3
"""
prime_compiler.py — Transformer Geométrico de Grafos de Primos
SPDX-License-Identifier: GPL-3.0-only

Converte código C/Pascal/Lua/Python simples em ARM32 assembly inline.
Cada operação tem um primo identificador. O produto de primos de um
caminho de código = fingerprint único desse caminho.

Uso:
    python3 prime_compiler.py --test
    python3 prime_compiler.py input.c
"""
import sys, re

# Mapa de primos para operações semânticas
PRIME = {
    'LOAD':2,'STORE':3,'ADD':5,'SUB':7,'MUL':11,'DIV':13,
    'AND':17,'OR':19,'XOR':23,'NOT':29,'SHL':31,'SHR':37,
    'CMP':41,'BRANCH':43,'CALL':47,'RET':53,'PUSH':59,'POP':61,
    'NOP':67,'MOVE':71,'EMA':73,'CRC':79,'SIN':83,'TORUS':89,
}

# Registradores ARM32 por primo
REG = {2:'r0',3:'r1',5:'r2',7:'r3',11:'r4',13:'r5',17:'r6',
       19:'r7',23:'r8',29:'r9',31:'r10',37:'r11',41:'r12',
       43:'sp',47:'lr',53:'pc'}

def crc32c(data):
    tab=[0]*256
    for i in range(256):
        v=i
        for _ in range(8): v=(v>>1)^0x82F63B78 if v&1 else v>>1
        tab[i]=v
    c=0xFFFFFFFF
    for b in data: c=(c>>8)^tab[(c^b)&0xFF]
    return (~c)&0xFFFFFFFF

def qema(old,inp): return ((old*49152+inp*16384)>>16)&0xFFFF
def qsin(x):
    while x>=411774: x-=411774
    neg=x>=205887
    if neg: x-=205887
    x2=(x*x)>>16; x3=(x2*x)>>16; x5=(x3*x2)>>16
    r=max(0,min(65535,x-(x3*10923>>16)+(x5*546>>16)))
    return 65535-r if neg else r

class PrimeGraph:
    def __init__(self):
        self.nodes=[]; self.edges=[]
    def add(self,op):
        p=PRIME.get(op.upper(),97)
        idx=len(self.nodes); self.nodes.append((op,p))
        if idx>0: self.edges.append((idx-1,idx))
        return p
    def product(self,a,b):
        from math import prod
        return prod(n[1] for n in self.nodes[a:b+1])
    def traverse(self,n,stride=3):
        from math import gcd
        if gcd(stride,n)!=1: stride=1
        v,pos=[],0
        for _ in range(n): v.append(pos); pos=(pos+stride)%n
        return v

class ARM32Gen:
    def __init__(self): self.out=['@ RAFAELIA Generated ARM32','@ SPDX-License-Identifier: GPL-3.0-only',
        '.arch armv7-a','.fpu neon-vfpv4','.text','.align 2','.global raf_generated',
        'raf_generated:','    push {r4, r5, r6, r7, lr}  @ p=11×13×17×19×47 snapshot']
    def load(self,dst,v):
        r=REG.get(PRIME.get(dst.upper(),2),'r0')
        try: iv=int(v); self.out.append(f'    mov {r}, #{iv}  @ {dst}={v}' if iv<256 else f'    ldr {r}, ={iv}  @ {dst}={v}')
        except: self.out.append(f'    ldr {r}, =g_{v}  @ {dst}={v}')
    def add(self,d,a,b): self.out.append(f'    add r0, r1, r2  @ {d}={a}+{b}')
    def ema_q16(self,d,a,b):
        self.out+=[f'    @ EMA Q16.16: {d}=0.75*{a}+0.25*{b}',
            '    umull r2, r3, r0, r4  @ 49152*old',
            '    umull r5, r6, r1, r7  @ 16384*in',
            '    add   r2, r2, r5',
            '    lsr   r0, r2, #16     @ >> 16']
    def ret(self): self.out.append('    pop {r4, r5, r6, r7, pc}  @ return+restore (atomic)')
    def asm(self): return '\n'.join(self.out)

def run42():
    C,H,chain=0x8000,0x8000,0
    trace=[]
    for cy in range(42):
        sv=qsin(cy*9804)
        C=qema(C,sv); H=qema(H,65535-sv)
        phi=((65535-H)*C)>>16
        trace.append(phi)
        chain=crc32c(bytes([phi&0xFF,(phi>>8)&0xFF,
                             chain&0xFF,(chain>>8)&0xFF,
                             (chain>>16)&0xFF,(chain>>24)&0xFF,0,0]))
    return chain,trace[-1]

def selftest():
    assert crc32c(b'123456789')==0xE3069283,f"CRC FAIL"
    print(f"CRC32C OK: 0xE3069283")
    chain,phi=run42()
    print(f"42-cycle chain: 0x{chain:08X}")
    print(f"phi_final:      0x{phi:04X}")
    g=PrimeGraph()
    g.add('LOAD');g.add('ADD');g.add('STORE')
    print(f"path product: {g.product(0,2)}")
    print(f"traverse 8 stride=3: {g.traverse(8)}")
    gen=ARM32Gen()
    gen.load('a','42'); gen.load('b','100'); gen.add('c','a','b')
    gen.ema_q16('s','s','c_in'); gen.ret()
    asm=gen.asm()
    print(f"ASM CRC32C: 0x{crc32c(asm.encode()):08X}")
    print("\n--- Generated ARM32 ---")
    print(asm)

if __name__=='__main__':
    if len(sys.argv)==1 or '--test' in sys.argv: selftest()
    else:
        src=open(sys.argv[1]).read()
        gen=ARM32Gen(); g=PrimeGraph()
        for ln in src.splitlines():
            if '=' in ln and '//' not in ln[:ln.find('=')]:
                parts=ln.strip().rstrip(';').split('=')
                if len(parts)==2: gen.load(parts[0].strip(),parts[1].strip()); g.add('STORE')
        gen.ret(); print(gen.asm())
EOF_PY
